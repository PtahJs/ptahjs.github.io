组合式 API ​

useApp() ​

用于访问 ptah 实例。

  • 示例

    js
    import { useApp } from "@ptahjs/core";

export default {
  name: "Header",
  setup() {
    const app = useApp();
    console.log(app);
    return {};
  },
};

useUI() ​

在模块组件中初始化使用。

  • 示例

    vue
    <template>
  <div :ref="item.id" class="ui-button">
    <button @click="onTestEventClick">按钮</button>
  </div>
</template>

<script>
import { useUI } from "@ptahjs/core";
export default defineComponent({
  name: "UiButton1",
  props: {
    item: {
      type: Object,
      default: () => {},
    },
  },

  setup(props) {
    useUI(props);

    // 测试点击事件
    const onTestEventClick = (e) => {};

    return { onTestEventClick };
  },
});
</script>

useExecuteEvent() ​

用于执行事件流程。

  • 示例

    js
    import { useExecuteEvent } from "@ptahjs/core";

const { run } = useExecuteEvent();

// 测试点击事件
const onTestEventClick = (e) => {
  run("click", props.item, () => {
    console.log("执行完成");
  });
};

usePropsFormModel() ​

用于配置项组件，处理扩展属性模型数据。

  • 示例

    jsx
    import { usePropsFormModel, settingProps } from "@ptahjs/core";

export default {
  name: "PropNumber",
  props: settingProps,
  setup(props) {
    const [model] = usePropsFormModel(props);
    return () => (
      <input
        type="number"
        vModel={model.value}
        placeholder={props.config.placeholder}
      />
    );
  },
};

useRenderListener() ​

监听图层渲染状态。

  • 示例

    js
    import { useRenderListener } from "@ptahjs/core";

useRenderListener({
  items: items,
  sceneId: id,
  loading: () => {
    console.log("加载中...");
  },
  complete: () => {
    console.log("加载完成！");
  },
});

useRenderPass() ​

监听渲染过程。

  • 示例

    js
    import { useRenderPass } from "@ptahjs/core";

useRenderPass((uiId) => {
  console.log("渲染中", uiId);
});

useRenderComplete() ​

监听当前场景渲染完成。

  • 示例

    js
    import { useRenderComplete } from "@ptahjs/core";

useRenderComplete(() => {
  console.log("渲染完成 - 当前场景");
});

useGlobalRenderComplete() ​

监听所有场景渲染完成。

  • 示例

    js
    import { useGlobalRenderComplete } from "@ptahjs/core";

useGlobalRenderComplete(() => {
  console.log("渲染完成 - 所有场景");
});
